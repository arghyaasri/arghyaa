def is_palindrome(word):
    return word == word[::-1]

print is_palindrome("abcba")
print is_palindrome("steven")

def rot(word,amount):
    rotated = ''
    for letter in word:
        letter = ord(letter) + amount
        rotated += chr(letter)
    return rotated
print rot("cheer",7)

def has_no_e(word):
    for letter in word:
        if letter == 'e':
            return False
    return True
